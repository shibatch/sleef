static int cpuSupportsVXE2() {
  static int ret = -1;
  if (ret == -1)
    ret = !!(getauxval(AT_HWCAP) & HWCAP_S390_VXRS_EXT2);
  return ret;
}

#define SUBST_IF_EXT1(funcvxe2) if (cpuSupportsVXE2()) p = funcvxe2;
#else
#define SUBST_IF_EXT1(funcvxe2)
#endif
